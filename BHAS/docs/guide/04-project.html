Poglavlje 4: ProjectController

U ovom poglavlju kreirat ćemo CRUD za projekte. Novi koncepti uključuju: statičke dropdown vrijednosti, uvjetno renderiranje badge-ova, textarea za višelinijski tekst i rad s nullable datumima.

Novi koncepti u ovom poglavlju:
  • SelectList od string niza (statičke vrijednosti)
  • Conditional badge rendering u Razor-u
  • TextAreaFor za višelinijski unos
  • Nullable DateTime u View-u

Korak 1: Kreiranje ProjectController

Lokacija datoteke: Controllers/ProjectController.cs 
using BHAS.DbFirst;
using System;
using System.Data.Entity;
using System.Linq;
using System.Web.Mvc;

namespace AS.MVCDemo.Controllers
{
    public class ProjectController : Controller
    {
        private static readonly string[] StatusValues = { "Planiran", "Aktivan", "Završen", "Otkazan" };

        // GET: Project
        public ActionResult Index()
        {
            using (var db = new StateStatisticsDBEntities())
            {
                var projects = db.Projects
                    .Include(x => x.Department)
                    .OrderBy(x => x.ProjectName)
                    .ToList();
                return View(projects);
            }
        }

        // GET: Project/Details/5
        public ActionResult Details(int id)
        {
            using (var db = new StateStatisticsDBEntities())
            {
                var project = db.Projects
                    .Include(x => x.Department)
                    .FirstOrDefault(x => x.ProjectID == id);

                if (project == null)
                    return HttpNotFound();

                return View(project);
            }
        }

        // GET: Project/Create
        public ActionResult Create()
        {
            using (var db = new StateStatisticsDBEntities())
            {
                PopulateDepartmentsSelectList(db, null);
                ViewBag.StatusList = new SelectList(StatusValues);
                return View();
            }
        }

        // POST: Project/Create
        [HttpPost]
        [ValidateAntiForgeryToken]
        public ActionResult Create([Bind(Include = "ProjectName,DepartmentID,Budget,StartDate,EndDate,Status,Description")] Project model)
        {
            if (ModelState.IsValid)
            {
                using (var db = new StateStatisticsDBEntities())
                {
                    model.CreatedDate = DateTime.Now;
                    model.ModifiedDate = DateTime.Now;
                    db.Projects.Add(model);
                    db.SaveChanges();
                    return RedirectToAction("Index");
                }
            }

            using (var db = new StateStatisticsDBEntities())
            {
                PopulateDepartmentsSelectList(db, model.DepartmentID);
                ViewBag.StatusList = new SelectList(StatusValues, model.Status);
            }
            return View(model);
        }

        // GET: Project/Edit/5
        public ActionResult Edit(int id)
        {
            using (var db = new StateStatisticsDBEntities())
            {
                var project = db.Projects.Find(id);
                if (project == null)
                    return HttpNotFound();

                PopulateDepartmentsSelectList(db, project.DepartmentID);
                ViewBag.StatusList = new SelectList(StatusValues, project.Status);
                return View(project);
            }
        }

        // POST: Project/Edit/5
        [HttpPost]
        [ValidateAntiForgeryToken]
        public ActionResult Edit([Bind(Include = "ProjectID,ProjectName,DepartmentID,Budget,StartDate,EndDate,Status,Description")] Project model)
        {
            if (ModelState.IsValid)
            {
                using (var db = new StateStatisticsDBEntities())
                {
                    model.ModifiedDate = DateTime.Now;
                    db.Entry(model).State = EntityState.Modified;
                    db.SaveChanges();
                    return RedirectToAction("Index");
                }
            }

            using (var db = new StateStatisticsDBEntities())
            {
                PopulateDepartmentsSelectList(db, model.DepartmentID);
                ViewBag.StatusList = new SelectList(StatusValues, model.Status);
            }
            return View(model);
        }

        // GET: Project/Delete/5
        public ActionResult Delete(int id)
        {
            using (var db = new StateStatisticsDBEntities())
            {
                var project = db.Projects
                    .Include(x => x.Department)
                    .FirstOrDefault(x => x.ProjectID == id);

                if (project == null)
                    return HttpNotFound();

                return View(project);
            }
        }

        // POST: Project/Delete/5
        [HttpPost, ActionName("Delete")]
        [ValidateAntiForgeryToken]
        public ActionResult DeleteConfirmed(int id)
        {
            using (var db = new StateStatisticsDBEntities())
            {
                var project = db.Projects.Find(id);
                if (project == null)
                    return HttpNotFound();

                db.Projects.Remove(project);
                db.SaveChanges();
                return RedirectToAction("Index");
            }
        }

        private void PopulateDepartmentsSelectList(StateStatisticsDBEntities db, int? selectedId)
        {
            var departments = db.Departments.OrderBy(x => x.DepartmentName).ToList();
            ViewBag.DepartmentID = new SelectList(departments, "DepartmentID", "DepartmentName", selectedId);
        }
    }
}

Novi koncepti u kontroleru

Statičke dropdown vrijednosti:
private static readonly string[] StatusValues = { "Planiran", "Aktivan", "Završen", "Otkazan" };

Umjesto da dohvaćamo statuse iz baze, definiramo ih kao konstantni niz stringova. Ovo je korisno za vrijednosti koje se rijetko mijenjaju.

Kreiranje SelectList-a od string niza:
ViewBag.StatusList = new SelectList(StatusValues) — Kreira dropdown bez create/edit (value i text su isti string)
ViewBag.StatusList = new SelectList(StatusValues, model.Status) — Kreira dropdown s predodabranom vrijednošću

Korak 2: Index View (Lista projekata)

Lokacija datoteke: Views/Project/Index.cshtml
Kreirajte folder Views/Project/, zatim dodajte Index.cshtml. 
@model List<BHAS.DbFirst.Project>
@{
    ViewBag.Title = "Projekti";
}

<h2>Projekti</h2>
<p>@Html.ActionLink("Kreiraj novi projekat", "Create", null, new { @class = "btn btn-primary" })</p>

<table class="table table-bordered table-striped table-hover">
    <thead class="table-dark">
        <tr>
            <th>Naziv projekta</th>
            <th>Odjel</th>
            <th>Budžet</th>
            <th>Početak</th>
            <th>Kraj</th>
            <th>Status</th>
            <th>Akcije</th>
        </tr>
    </thead>
    <tbody>
        @foreach (var item in Model)
        {
            <tr>
                <td>@item.ProjectName</td>
                <td>@(item.Department != null ? item.Department.DepartmentName : "-")</td>
                <td>@item.Budget.ToString("N2")</td>
                <td>@item.StartDate.ToString("dd.MM.yyyy")</td>
                <td>@(item.EndDate.HasValue ? item.EndDate.Value.ToString("dd.MM.yyyy") : "-")</td>
                <td>
                    @{
                        var badgeClass = item.Status == "Aktivan" ? "bg-success"
                            : item.Status == "Završen" ? "bg-secondary"
                            : item.Status == "Otkazan" ? "bg-danger"
                            : "bg-warning text-dark";
                    }
                    <span class="badge @badgeClass">@item.Status</span>
                </td>
                <td>
                    @Html.ActionLink("Detalji", "Details", new { id = item.ProjectID }, new { @class = "btn btn-sm btn-info" })
                    @Html.ActionLink("Uredi", "Edit", new { id = item.ProjectID }, new { @class = "btn btn-sm btn-warning" })
                    @Html.ActionLink("Obriši", "Delete", new { id = item.ProjectID }, new { @class = "btn btn-sm btn-danger" })
                </td>
            </tr>
        }
    </tbody>
</table>
Uvjetno renderiranje badge-ova:
Unutar @{ ... } bloka pišemo C# kod koji se izvršava na serveru:
@{
    var badgeClass = item.Status == "Aktivan" ? "bg-success"
        : item.Status == "Završen" ? "bg-secondary"
        : item.Status == "Otkazan" ? "bg-danger"
        : "bg-warning text-dark";
}
Ovo koristi ulančane ternarne operatore za odabir Bootstrap badge klase na osnovu statusa projekta.
Rezultat se koristi u: <span class="badge @badgeClass">@item.Status</span>

Nullable datum:
@(item.EndDate.HasValue ? item.EndDate.Value.ToString("dd.MM.yyyy") : "-")
EndDate je DateTime? (nullable) jer projekat možda još nema datum završetka.

Novi HTML helperi u ovom View-u:

1. DropDownList sa eksplicitnim cast-om:
@Html.DropDownList("Status", (SelectList)ViewBag.StatusList, "-- odaberi status --", ...)
Ovdje eksplicitno kastujemo ViewBag.StatusList u SelectList jer se ime DropDownList-a ("Status") razlikuje od imena ViewBag svojstva (StatusList).

2. TextAreaFor:
@Html.TextAreaFor(m => m.Description, new { @class = "form-control", rows = 3 })
Generira <textarea> tag umjesto <input>. Parametar rows = 3 postavlja visinu na 3 reda.

Null-conditional operator:
@(Model.Department?.DepartmentName ?? "-")

?. (null-conditional) — Ako je Department null, ne baca grešku nego vraća null.
?? (null-coalescing) — Ako je rezultat null, koristi "-" kao default vrijednost.

Ovo je kraći zapis od: Model.Department != null ? Model.Department.DepartmentName : "-"

Šta ste naučili u ovom poglavlju

  • Statički dropdown — new SelectList(string[]) za fiksne vrijednosti
  • Badge rendering — Ulančani ternarni operatori za dinamičke CSS klase
  • TextAreaFor — Za višelinijski unos teksta
  • Eksplicitni cast ViewBag-a — (SelectList)ViewBag.StatusList kada se ime razlikuje
  • Null-conditional operator — ?. i ?? za elegantan rad s null vrijednostima
  • Nullable DateTime — HasValue provjera za opcione datume
