Poglavlje 2: DepartmentController

U ovom poglavlju kreirat ćemo kompletni CRUD za upravljanje odjelima, uključujući AJAX parcijalni view za prikaz zaposlenika odjela. Ovo je najdetaljnije poglavlje jer uvodi sve ključne koncepte koje ćemo ponavljati u narednim poglavljima.

Sadržaj poglavlja: Controller | Index View | Details View | Create View | Edit View | Delete View | Partial View + AJAX

Korak 1: Kreiranje DepartmentController

Lokacija datoteke: Controllers/DepartmentController.cs
Desni klik na folder Controllers → Add → Controller → MVC 5 Controller - Empty → ime: DepartmentController

Ovaj kontroler sadrži sve CRUD akcije plus jednu akciju za AJAX učitavanje zaposlenika. Otkucajte kompletan kod:

using BHAS.DbFirst;
using System;
using System.Data.Entity;
using System.Linq;
using System.Web.Mvc;

namespace AS.MVCDemo.Controllers
{
    public class DepartmentController : Controller
    {
        // GET: Department
        public ActionResult Index()
        {
            using (var db = new StateStatisticsDBEntities())
            {
                var departments = db.Departments.ToList();
                return View(departments);
            }
        }

        // GET: Department/Details/5
        public ActionResult Details(int id)
        {
            using (var db = new StateStatisticsDBEntities())
            {
                var department = db.Departments
                    .Include(x => x.Employees)
                    .Include(x => x.Employee)
                    .FirstOrDefault(x => x.DepartmentID == id);

                if (department == null)
                    return HttpNotFound();

                return View(department);
            }
        }

        // GET: Department/Create
        public ActionResult Create()
        {
            using (var db = new StateStatisticsDBEntities())
            {
                PopulateManagersSelectList(db, null);
                return View();
            }
        }

        // POST: Department/Create
        [HttpPost]
        [ValidateAntiForgeryToken]
        public ActionResult Create([Bind(Include = "DepartmentName,DepartmentCode,Budget,ManagerID")] Department model)
        {
            if (ModelState.IsValid)
            {
                using (var db = new StateStatisticsDBEntities())
                {
                    model.CreatedDate = DateTime.Now;
                    model.ModifiedDate = DateTime.Now;
                    db.Departments.Add(model);
                    db.SaveChanges();
                    return RedirectToAction("Index");
                }
            }

            using (var db = new StateStatisticsDBEntities())
            {
                PopulateManagersSelectList(db, model.ManagerID);
            }
            return View(model);
        }

        // GET: Department/Edit/5
        public ActionResult Edit(int id)
        {
            using (var db = new StateStatisticsDBEntities())
            {
                var department = db.Departments.Find(id);
                if (department == null)
                    return HttpNotFound();

                PopulateManagersSelectList(db, department.ManagerID);
                return View(department);
            }
        }

        // POST: Department/Edit/5
        [HttpPost]
        [ValidateAntiForgeryToken]
        public ActionResult Edit([Bind(Include = "DepartmentID,DepartmentName,DepartmentCode,Budget,ManagerID")] Department model)
        {
            if (ModelState.IsValid)
            {
                using (var db = new StateStatisticsDBEntities())
                {
                    model.ModifiedDate = DateTime.Now;
                    db.Entry(model).State = EntityState.Modified;
                    db.SaveChanges();
                    return RedirectToAction("Index");
                }
            }

            using (var db = new StateStatisticsDBEntities())
            {
                PopulateManagersSelectList(db, model.ManagerID);
            }
            return View(model);
        }

        // GET: Department/Delete/5
        public ActionResult Delete(int id)
        {
            using (var db = new StateStatisticsDBEntities())
            {
                var department = db.Departments.Find(id);
                if (department == null)
                    return HttpNotFound();

                return View(department);
            }
        }

        // POST: Department/Delete/5
        [HttpPost, ActionName("Delete")]
        [ValidateAntiForgeryToken]
        public ActionResult DeleteConfirmed(int id)
        {
            using (var db = new StateStatisticsDBEntities())
            {
                try
                {
                    var department = db.Departments.Find(id);
                    if (department == null)
                        return HttpNotFound();

                    db.Departments.Remove(department);
                    db.SaveChanges();
                    return RedirectToAction("Index");
                }
                catch (Exception)
                {
                    TempData["Error"] = "Nije moguće obrisati odjel jer ima zaposlenih. Prvo premjestite ili obrišite zaposlene.";
                    return RedirectToAction("Details", new { id });
                }
            }
        }

        // GET: Department/EmployeesPartial?departmentId=5
        [HttpGet]
        public ActionResult EmployeesPartial(int departmentId)
        {
            using (var db = new StateStatisticsDBEntities())
            {
                var employees = db.Employees
                    .Where(x => x.DepartmentID == departmentId)
                    .OrderBy(x => x.LastName)
                    .ThenBy(x => x.FirstName)
                    .ToList();

                return PartialView("_EmployeesTable", employees);
            }
        }

        private void PopulateManagersSelectList(StateStatisticsDBEntities db, int? selectedEmployeeId)
        {
            var employees = db.Employees
                .OrderBy(x => x.LastName)
                .ThenBy(x => x.FirstName)
                .Select(x => new
                {
                    x.EmployeeID,
                    FullName = x.LastName + " " + x.FirstName
                })
                .ToList();

            ViewBag.ManagerID = new SelectList(employees, "EmployeeID", "FullName", selectedEmployeeId);
        }
    }
}

Detaljno objašnjenje svake akcije

Index() — Prikaz svih odjela

db.Departments.ToList() dohvaća sve zapise iz tabele Departments.
return View(departments) prosljeđuje listu odjela kao Model u View.
View za ovu akciju se traži na lokaciji: Views/Department/Index.cshtml

Details(int id) — Prikaz detalja jednog odjela

.Include(x => x.Employees) koristi eager loading da zajedno s odjelom dohvati i sve zaposlenike.
.Include(x => x.Employee) dohvaća menadžera odjela (navigacijsko svojstvo Employee predstavlja FK na menadžera).
FirstOrDefault(x => x.DepartmentID == id) pronalazi odjel po ID-u.
HttpNotFound() vraća HTTP 404 ako odjel ne postoji.

Create() GET i POST — Kreiranje novog odjela

GET akcija prikazuje praznu formu. Prije toga poziva PopulateManagersSelectList da napuni dropdown listu menadžera.

POST akcija prima podatke iz forme:
  • [HttpPost] — Ova akcija reagira samo na POST zahtjeve
  • [ValidateAntiForgeryToken] — Štiti od CSRF napada (Cross-Site Request Forgery)
  • [Bind(Include = "...")] — Dozvoljava samo navedena svojstva da budu popunjena iz forme (sigurnosna mjera)
  • ModelState.IsValid — Provjerava da li su svi validacijski uvjeti ispunjeni
  • db.Departments.Add(model) — Dodaje novi zapis u DbSet
  • db.SaveChanges() — Sprema promjene u bazu podataka
  • RedirectToAction("Index") — Preusmjerava korisnika na listu odjela

Edit() GET i POST — Uređivanje odjela

Razlika od Create:
  • db.Departments.Find(id) — Pronalazi postojeći zapis
  • POST verzija Edit-a koristi db.Entry(model).State = EntityState.Modified umjesto Add()
  • U [Bind] atributu, Edit uključuje i DepartmentID jer ažuriramo postojeći zapis

Delete() i DeleteConfirmed() — Brisanje odjela

Zašto dvije metode? Ne možemo imati dvije metode s istim imenom i istim parametrima.
[HttpPost, ActionName("Delete")] govori MVC-u da se DeleteConfirmed metoda poziva kada korisnik pošalje POST zahtjev na URL /Department/Delete/5.

try-catch blok hvata grešku ako pokušamo obrisati odjel koji ima zaposlene (zbog FK constraint-a u bazi).
TempData["Error"] sprema poruku greške koja će se prikazati na Details stranici nakon preusmjeravanja.

EmployeesPartial() — AJAX akcija

PartialView("_EmployeesTable", employees) vraća samo dio HTML-a (bez Layout stranice).
Ovo se koristi za AJAX pozive — JavaScript šalje zahtjev, prima HTML fragment i ubacuje ga u stranicu bez potpunog osvježavanja.

PopulateManagersSelectList() — Pomoćna metoda

Ova privatna metoda dohvaća sve zaposlenike, formatira ih kao "Prezime Ime" i kreira SelectList objekt koji se sprema u ViewBag.ManagerID.
new SelectList(employees, "EmployeeID", "FullName", selectedEmployeeId)
Parametri: kolekcija, value field, text field, trenutno odabrana vrijednost.

@section scripts {

        $(function () {

}

Objašnjenje koda

Element Objašnjenje
@model List<...Department> Model direktiva — Deklarira tip podataka koji View očekuje. Ovo omogućava IntelliSense i type-checking.
@foreach (var item in Model) Iterira kroz listu odjela. Model je varijabla koja sadrži podatke proslijeđene iz kontrolera.
@item.Budget.ToString("N2") Formatira broj s 2 decimalna mjesta i separatorom hiljada (npr. 1,500,000.00).
data-id="@item.DepartmentID" HTML5 data atribut koji sprema ID odjela. JavaScript ga čita pomoću $(this).data("id").
@Html.ActionLink("Detalji", "Details", new { id = item.DepartmentID }, ...) Kreira link koji vodi na /Department/Details/5. new { id = item.DepartmentID } su route values.
@section scripts { ... } Definira sekciju koja se ubacuje na mjesto @RenderSection("scripts") u Layout stranici. Ovo osigurava da se skripte učitaju nakon jQuery-ja.
$.get("/Department/EmployeesPartial", { departmentId: departmentId }) jQuery AJAX GET zahtjev koji poziva EmployeesPartial akciju i prima HTML fragment.
AJAX obrazac u ovom View-u:
1. Korisnik klikne na red u tabeli
2. JavaScript čita data-id atribut iz kliknutog reda
3. Šalje AJAX zahtjev na /Department/EmployeesPartial?departmentId=X
4. Server vraća HTML tabelu zaposlenika (partial view)
5. JavaScript ubacuje primljeni HTML u #employeesHost div

Objašnjenje novih koncepata

Element Objašnjenje
@model BHAS.DbFirst.Department Ovaj View prima jedan Department objekat (ne listu).
TempData["Error"] TempData je dictionary koji čuva podatke kroz jedno preusmjeravanje. Koristimo ga za prikaz greške nakon neuspjelog brisanja.
<dl class="row"> HTML definition list s Bootstrap grid klasama za tabularni prikaz podataka.
Model.Employee Navigacijsko svojstvo prema menadžeru. Može biti null ako menadžer nije dodijeljen.
Model.CreatedDate.HasValue Provjera da li nullable DateTime ima vrijednost. Ako nema, prikazujemo "-".
TempData vs ViewBag vs Session:
  • ViewBag — Živi samo tokom jednog zahtjeva (request). Ne preživljava redirect.
  • TempData — Živi tokom jednog zahtjeva + jednog preusmjeravanja. Savršeno za poruke nakon RedirectToAction.
  • Session — Živi tokom cijele korisničke sesije. Koristiti oprezno.

Objašnjenje forme i HTML helpera

Element Objašnjenje
@using (Html.BeginForm()) Generira <form> tag koji šalje POST zahtjev na istu akciju (Create). using osigurava zatvaranje </form> taga.
@Html.AntiForgeryToken() Generira skriveno polje s CSRF tokenom. Odgovara [ValidateAntiForgeryToken] atributu na POST akciji.
@Html.ValidationSummary(true, "", ...) Prikazuje listu svih validacijskih grešaka. true znači da prikazuje samo greške na nivou modela (ne i za pojedinačna polja).
@Html.LabelFor(m => m.DepartmentName, "Naziv odjela") Generira <label> tag za svojstvo DepartmentName s prilagođenim tekstom.
@Html.EditorFor(m => m.DepartmentName, new { htmlAttributes = new { @class = "form-control" } }) Generira odgovarajući <input> tag na osnovu tipa svojstva. htmlAttributes dodaje CSS klase.
@Html.ValidationMessageFor(m => m.DepartmentName, "") Prikazuje validacijsku poruku greške za konkretno polje.
@Html.DropDownList("ManagerID", null, "-- odaberi menadžera --", ...) Generira <select> tag. Koristi ViewBag.ManagerID (SelectList) koji smo postavili u kontroleru. Treći parametar je placeholder opcija.
@Scripts.Render("~/bundles/jqueryval") Učitava jQuery Validate i Unobtrusive Validation skripte za klijentsku validaciju.
Kako DropDownList i ViewBag rade zajedno:
1. U kontroleru: ViewBag.ManagerID = new SelectList(employees, "EmployeeID", "FullName", selectedId)
2. U View-u: @Html.DropDownList("ManagerID", null, ...)

Kada ime DropDownList-a ("ManagerID") odgovara imenu ViewBag svojstva (ViewBag.ManagerID), MVC automatski koristi tu SelectList kolekciju. Zato je drugi parametar null.

Ključna razlika Edit od Create:

Objašnjenje Delete obrasca

Zašto koristimo formu za brisanje?
Brisanje podataka je destruktivna operacija i mora se slati kao POST zahtjev (ne GET). GET zahtjevi mogu biti slučajno pozvani od strane crawlera, prefetch-a, ili čak korisnika koji klikne na link.

Html.BeginForm("Delete", "Department", FormMethod.Post) — Eksplicitno navodi akciju, kontroler i HTTP metodu. Ovo je potrebno jer Delete View nije na istom URL-u kao POST akcija.

Korisnik vidi podatke odjela i mora kliknuti "Da, obriši" da potvrdi brisanje.

Korak 7: _EmployeesTable Partial View

Lokacija datoteke: Views/Department/_EmployeesTable.cshtml
Konvencija: Partial view-ovi počinju sa _ (donja crta). 
@model List<BHAS.DbFirst.Employee>

@if (Model == null || !Model.Any())
{
    <div class="alert alert-info">Nema zaposlenih u ovom departmanu.</div>
}
else
{
    <table class="table table-bordered table-striped">
        <thead class="table-secondary">
            <tr>
                <th>Ime</th>
                <th>Prezime</th>
                <th>Pozicija</th>
                <th>Plata</th>
            </tr>
        </thead>
        <tbody>
            @foreach (var emp in Model)
            {
                <tr>
                    <td>@emp.FirstName</td>
                    <td>@emp.LastName</td>
                    <td>@emp.Position</td>
                    <td>@emp.Salary.ToString("N2")</td>
                </tr>
            }
        </tbody>
    </table>
    <p class="text-muted">Ukupno zaposlenih: @Model.Count</p>
}
Šta je Partial View?
Partial View je View koji nema Layout. Renderira samo svoj sadržaj (HTML fragment) bez <html>, <head>, <body> tagova.

Koristi se za:
  • AJAX pozive — Server vraća samo dio HTML-a koji JavaScript ubacuje u stranicu
  • Ponovnu upotrebu — Isti partial view se može koristiti na više stranica
  • Organizaciju — Razdvajanje velikih View-ova na manje dijelove

U kontroleru ga vraćamo s: return PartialView("_EmployeesTable", employees);

Šta ste naučili u ovom poglavlju

  • CRUD obrazac — Index (lista), Details (detalji), Create (kreiranje), Edit (uređivanje), Delete (brisanje)
  • Entity Framework — using (var db = new StateStatisticsDBEntities()) za pristup bazi
  • Include() — Eager loading za dohvatanje povezanih entiteta
  • [HttpPost], [ValidateAntiForgeryToken], [Bind] — Atributi za sigurnost i kontrolu POST akcija
  • ViewBag i SelectList — Za popunjavanje DropDownList kontrola u formama
  • Html.BeginForm, EditorFor, LabelFor, ValidationMessageFor — HTML helperi za kreiranje formi
  • HiddenFor — Skriveno polje za ID kod Edit akcije
  • TempData — Za prosljeđivanje poruka kroz redirect
  • Partial View — HTML fragment bez Layout-a, za AJAX ili ponovnu upotrebu
  • AJAX sa jQuery — $.get() za asinhrono učitavanje sadržaja bez osvježavanja stranice
  • ActionName atribut — Za mapiranje metode s drugačijim imenom na istu URL rutu
